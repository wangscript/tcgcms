DZ.NET论坛商场批量加入地区的脚本发布：2009年10月03日 浏览： 次

DECLARE @AREASTR NVARCHAR(2000)
DECLARE @TEMP NVARCHAR(10)
DECLARE @AREATE_BIG NVARCHAR(10)
DECLARE @COUNT INT

SET @AREATE_BIG = '驻道观'
SET @AREASTR = '道观、石家寨、桐子岗、油麻岭、汉子山、王家河'

WHILE(CHARINDEX('、',@AREASTR)>0)
BEGIN

SET @COUNT = 0
SET @TEMP = SUBSTRING(@AREASTR ,0,CHARINDEX('、',@AREASTR))


SELECT @COUNT =COUNT(1) FROM dbo.dnt_locations WHERE CITY = @TEMP AND [STATE] = @AREATE_BIG

IF(@COUNT=0)
BEGIN
 INSERT INTO [TCG_BBS].[dbo].[dnt_locations]
           ([city]
           ,[state]
           ,[country]
           ,[zipcode])
     VALUES
           (@TEMP
           ,@AREATE_BIG
           ,'新洲区'
           ,'')
END
SET @AREASTR = SUBSTRING(@AREASTR ,CHARINDEX('、',@AREASTR)+1,LEN(@AREASTR)-1)
END

IF(@AREASTR!='')
BEGIN
 SET @COUNT = 0
 SELECT @COUNT =COUNT(1) FROM dbo.dnt_locations WHERE CITY = @AREASTR AND [STATE] = @AREATE_BIG

 SELECT @COUNT
 IF(@COUNT=0)
 BEGIN
  INSERT INTO [TCG_BBS].[dbo].[dnt_locations]
           ([city]
           ,[state]
           ,[country]
           ,[zipcode])
   VALUES
      (@AREASTR
      ,@AREATE_BIG
      ,'新洲区'
      ,'')
 END
END

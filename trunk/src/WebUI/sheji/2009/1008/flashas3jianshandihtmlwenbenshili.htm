flashas3简单的html文本实例发布：2009年10月10日 浏览： 次

图如下：

代码：

package com{
import flash.display.Sprite;
import flash.text.TextField;
import flash.text.TextFieldType;
public class HtmlText extends Sprite{
   private var tx:TextField;
   public function HtmlText(){
    tx = new TextField();
    addChild(tx);
    tx.x=tx.y=100;
    tx.autoSize = "left";
    tx.htmlText = "现在显示的是html文本，这是<b>加粗</b>，这是<font color='#FF0000'>红色的文字</font>。";
   }
}
}
